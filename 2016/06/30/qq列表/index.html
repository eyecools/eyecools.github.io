可收缩分组表格

这是一个简单的课收缩的的分组表格，点击其中的的一个cell就会展开相应的列表，再次点击时会进行列表收缩的操作，下面是效果图和源码

点击后的效果：

@property(nonatomic,strong)NSMutableArray *dataArray;
//用可变字典记录收缩状态
@property(nonatomic,strong)NSMutableDictionary *tagDic;
@property(nonatomic,assign)BOOL re;
@end
@implementation ListViewController

- (void)viewDidLoad {
    [super viewDidLoad];
    //数据源
    self.dataArray = [NSMutableArray arrayWithObjects:@[@"盖伦",@"赵信",@"德莱厄斯"],@[@"关羽",@"张飞",@"赵云"],@[@"诸葛亮",@"庞统",@"司马懿"],@[@"杨老师",@"王老师",@"苍老师"], nil];

    //初始化用于标记某一个section的折叠状态的字典
    self.tagDic = [NSMutableDictionary dictionary];

    self.re = YES;
}

-(UIView *)tableView:(UITableView *)tableView viewForHeaderInSection:(NSInteger)section{
    UIButton *btn = [UIButton buttonWithType:(UIButtonTypeCustom)];
    btn.backgroundColor = [UIColor magentaColor];
    [btn setTitle:[NSString stringWithFormat:@"第%ld组",section] forState:(UIControlStateNormal)];
    [btn addTarget:self action:@selector(Btn:) forControlEvents:(UIControlEventTouchUpInside)];
    btn.tag = section;
    return btn;
}
-(void)Btn:(UIButton *)btn{
    //如果被点击的section状态为“0（打开）”
    if ([[self.tagDic valueForKey:[NSString stringWithFormat:@"%ld",btn.tag]] isEqualToString:@"0"]) {
        //那么将其状态改为“1（关闭）”
        [self.tagDic setValue:@"1" forKey:[NSString stringWithFormat:@"%ld",btn.tag]];
    }else{
        //否则改为“0（打开）”
        [self.tagDic setValue:@"0" forKey:[NSString stringWithFormat:@"%ld",btn.tag]];
    }
    //刷新section
    [self.tableView reloadSections:[NSIndexSet indexSetWithIndex:btn.tag] withRowAnimation:(UITableViewRowAnimationFade)];
}
- (NSInteger)numberOfSectionsInTableView:(UITableView *)tableView {
    return self.dataArray.count;
}

- (NSInteger)tableView:(UITableView *)tableView numberOfRowsInSection:(NSInteger)section {
    //valueForKey：取值
    //如果所选分区状态为“0（打开）”
    if ([[self.tagDic valueForKey:[NSString stringWithFormat:@"%ld",section]] isEqualToString:@"0"]) {
        return [self.dataArray[section]count];
    }else{
        return 0;
    }
}
- (UITableViewCell *)tableView:(UITableView *)tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath {
    UITableViewCell *cell = [tableView dequeueReusableCellWithIdentifier:@"reuseIdentifier"];
    if (cell == nil) {
        cell = [[UITableViewCell alloc]initWithStyle:(UITableViewCellStyleDefault) reuseIdentifier:@"reuseIdentifier"
                ];
    }
    //赋予每行内容（对应分区的对应行）
    cell.textLabel.text = self.dataArray[indexPath.section][indexPath.row];

    return cell;
}
-(CGFloat)tableView:(UITableView *)tableView heightForHeaderInSection:(NSInteger)section{
    return 50;
}
